Propertiesedit

Exampleedit

   "comment" : {
      "id" : "string",
      "message" : "string",
      "user_id" : "string"
   },

      "created_time" : "2019-01-01T00:00:00Z",
      "modified_time" : "2019-01-01T00:00:00Z",
      "version" : "string"
   }
}
