RunnerOverviewedit

Information about all of the runners.

Propertiesedit

runners (array[RunnerInfo], required)
List of runners

Exampleedit

{
   "runners" : [
      {
         "build_info" : {
            "commit_hash" : "string",
            "version" : "string"
         },
         "connected" : true,
         "containers" : [
            {
               "container_name" : "string",
               "container_set_name" : "string"
            }
         ],
         "features" : {
            "some_property" : true
         },
         "health_checks" : {
            "has_containers" : true,
            "is_connected_to_zk" : true,
            "is_docker_healthy" : true
         },
         "healthy" : true,
         "host_ip" : "string",
         "public_hostname" : "string",
         "region" : "string",
         "roles" : [
            {
               "role_name" : "string"
            }
         ],
         "runner_id" : "string",
         "runner_name" : "string",
         "tags" : {
            "some_property" : "string"
         },
         "zone" : "string"
      }
   ]
}
