// Span Near
QueryBuilder qb = spanNearQuery()
    .clause(spanTermQuery("field","value1"))    
    .clause(spanTermQuery("field","value2"))    
    .clause(spanTermQuery("field","value3"))    
    .slop(12)                                   
    .inOrder(false)                             
    .collectPayloads(false);
