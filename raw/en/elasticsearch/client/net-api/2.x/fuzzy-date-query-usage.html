Fuzzy Date Query Usageedit

Fluent DSL exampleedit

q
.FuzzyDate(c => c
    .Name("named_query")
    .Boost(1.1)
    .Field(p => p.Description)
    .Fuzziness(TimeSpan.FromDays(2))
    .Value(Project.Instance.StartedOn)
    .MaxExpansions(100)
    .PrefixLength(3)
    .Rewrite(MultiTermQueryRewrite.ConstantScore)
    .Transpositions()
)

Object Initializer syntax exampleedit

new FuzzyDateQuery
{
    Name = "named_query",
    Boost = 1.1,
    Field = "description",
    Fuzziness = TimeSpan.FromDays(2),
    Value = Project.Instance.StartedOn,
    MaxExpansions = 100,
    PrefixLength = 3,
    MultiTermQueryRewrite = MultiTermQueryRewrite.ConstantScore,
    Transpositions = true
}
