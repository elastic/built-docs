Geo Shape Multi Line String Query Usage

Fluent DSL example

edit
q
.GeoShapeMultiLineString(c => c
    .Name("named_query")
    .Boost(1.1)
    .Field(p => p.LocationShape)
    .Coordinates(MultiLineStringCoordinates)
    .Relation(GeoShapeRelation.Intersects)
)

Object Initializer syntax example

edit
new GeoShapeMultiLineStringQuery
{
    Name = "named_query",
    Boost = 1.1,
    Field = Field<Project>(p => p.LocationShape),
    Shape = new MultiLineStringGeoShape(MultiLineStringCoordinates),
    Relation = GeoShapeRelation.Intersects,
}
