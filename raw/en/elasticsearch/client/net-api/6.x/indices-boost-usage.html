Indices Boost Usageedit

Fluent DSL exampleedit

s => s
.IndicesBoost(b => b
    .Add("project", 1.4)
    .Add("devs", 1.3)
)
.Query(q => q
    .MatchAll()
)

Object Initializer syntax exampleedit

new SearchRequest<Project>
{
    IndicesBoost = new Dictionary<IndexName, double>
    {
        { "project", 1.4 },
        { "devs", 1.3 }
    },
    Query = new MatchAllQuery()
}

Example json output.

{
  "indices_boost": [
    {
      "project": 1.4
    },
    {
      "devs": 1.3
    }
  ],
  "query": {
    "match_all": {}
  }
}
